/****************************************************************** 🚀 CODEIQ AI - SINGLE FILE FULL STACK FRONTEND + BACKEND + AI + CACHE + MEMORY + ANALYTICS ******************************************************************/ import express from "express"; import cors from "cors"; import dotenv from "dotenv"; import OpenAI from "openai"; dotenv.config(); const app = express(); app.use(cors()); app.use(express.json({ limit: "10mb" })); const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY, }); // ===== CONFIG ===== const CACHE_TTL = 1000 * 60 * 5; const RATE_LIMIT = 150; const PORT = process.env.PORT || 3000; // ===== MEMORY SYSTEM ===== let users = {}; let cache = {}; let conversations = {}; let analytics = {}; let requestMap = {}; // ===== USER ===== function getUser(userId) { if (!users[userId]) { users[userId] = { userId, queries: 20, usage: 0, plan: "free", createdAt: Date.now() }; } return users[userId]; } // ===== RATE LIMIT ===== app.use((req, res, next) => { const ip = req.ip || req.connection.remoteAddress || "unknown"; requestMap[ip] = (requestMap[ip] || 0) + 1; if (requestMap[ip] > RATE_LIMIT) { return res.json({ reply: "⚠️ Too many requests. Try after 15 min." }); } setTimeout(() => { requestMap[ip] = 0 }, 15 * 60 * 1000); next(); }); // ===== CHAT API ===== app.post("/chat", async (req, res) => { try { const { userId, message } = req.body; if (!message || message.length < 2) { return res.json({ reply: "⚠️ Enter valid input", remaining: 0 }); } const user = getUser(userId); if (user.queries <= 0) { return res.json({ reply: "❌ Trial completed. Upgrade required.", remaining: 0 }); } // CACHE CHECK if (cache[message] && Date.now() - cache[message].time < CACHE_TTL) { return res.json(cache[message].data); } user.queries--; user.usage++; // MEMORY if (!conversations[userId]) conversations[userId] = []; conversations[userId].push({ role: "user", content: message }); const messages = [ { role: "system", content: `You are a strict medical coding AI. Rules: - Always give ICD & CPT - No guessing - Ask if unclear - Keep short Format: ✅ ICD: ✅ CPT: 🧠 Reason: ⚠️ Check: Confidence:` }, ...conversations[userId].slice(-6) ]; // RETRY LOGIC let ai; for (let i = 0; i < 2; i++) { try { ai = await openai.chat.completions.create({ model: "gpt-4o-mini", messages, temperature: 0.3 }); break; } catch (err) { console.error("OpenAI Error:", err.message); if (i === 1) throw err; } } let reply = ai?.choices?.[0]?.message?.content || "⚠️ Try again"; conversations[userId].push({ role: "assistant", content: reply }); // VALIDATION if (!reply.includes("ICD")) { reply = "⚠️ Low confidence. Please provide more details."; } // ANALYTICS analytics[message] = (analytics[message] || 0) + 1; const result = { reply, remaining: user.queries, usage: user.usage }; cache[message] = { data: result, time: Date.now() }; res.json(result); } catch (err) { console.error("Chat Error:", err.message); res.json({ reply: "⚠️ Server error. Check API Key.", remaining: 0 }); } }); // ===== FILE ANALYSIS ===== app.post("/upload", async (req, res) => { try { const { text } = req.body; if (!text) return res.json({ reply: "No content provided" }); const ai = await openai.chat.completions.create({ model: "gpt-4o-mini", messages: [ { role: "system", content: "Extract ICD and CPT codes clearly from medical text. Format: ICD: xxx, CPT: xxx, Reason: xxx" }, { role: "user", content: text } ] }); res.json({ reply: ai.choices[0].message.content }); } catch (err) { console.error("Upload Error:", err.message); res.json({ reply: "File analysis error" }); } }); // ===== UPGRADE ===== app.post("/upgrade", (req, res) => { const { userId, plan } = req.body; const user = getUser(userId); const plans = { "199": 100, "499": 300, "999": 700, "2999": 999999 }; user.plan = plan; user.queries = plans || 20; res.json({ success: true, user }); }); // ===== DASHBOARD ===== app.get("/dashboard/:id", (req, res) => { const user = getUser(req.params.id); res.json({ plan: user.plan, queries: user.queries, usage: user.usage }); }); // ===== ANALYTICS ===== app.get("/analytics", (req, res) => { res.json(analytics); }); // ===== FRONTEND UI ===== app.get("/", (req, res) => { res.send(` CodeIQ AI

🚀 CodeIQ AI - Medical Coding Assistant

Remaining Queries
20
Total Usage
0
Plan
FREE
`); }); // ===== START SERVER ===== app.listen(PORT, () => { console.log(\` 🚀 CodeIQ AI Running Successfully! 📍 Local: http://localhost:\${PORT} 🔑 Make sure OPENAI_API_KEY is set in.env file \`); });
